\begin{frame}
\frametitle{Get \LaTeX}
\begin{enumerate}
\item

Install \TeXLive\footnote{\url{http://tug.org/texlive}} or \MacTeX\footnote{\url{http://www.tug.org/mactex/2009}}.

\item

Install \TeXworks\footnote{\url{http://code.google.com/p/texworks}}.

\end{enumerate}
\end{frame}

\begin{frame}
\frametitle{Type \LaTeX}
\begin{enumerate}
\item

Open \TeXworks.

\item

Set the typeset to \texttt{pdfLaTeX}.

\item

Save a new file as \texttt{hello.tex}.

\end{enumerate}
\end{frame}

\begin{frame}[fragile]
\frametitle{Type \LaTeX}
\begin{enumerate}
% Continue enumeration
% From http://www.personal.ceu.hu/tex/counters.htm
\setcounter{enumi}{3}
\item
Type the following code:
\begin{verbatim}
\documentclass{article}
\title{Hello World}
\begin{document}
\maketitle
\end{document}
\end{verbatim}

\end{enumerate}
\end{frame}

\begin{frame}
\frametitle{Compile \LaTeX}
\begin{enumerate}

\item
% Vertical alignment by John J. Camilleri (http://johnjcamilleri.com/2010/05/vertical-alignment-of-inline-images-in-latex/)
Click the green typeset button (\raisebox{-1.6mm}{\includegraphics[scale=0.5]{runtool.png}}).

\item
The {\LaTeX} code compiles to a PDF file.

\item
A preview appears.

\end{enumerate}
\end{frame}
